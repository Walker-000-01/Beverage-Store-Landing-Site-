        <script>
            (function () {
                const STORAGE_KEY = 'millanes_testimonials';
                const form = document.getElementById('test-form');
                const list = document.getElementById('testimonials-list');

                function renderStars(n) {
                    n = Number(n) || 0;
                    return 'â˜…'.repeat(n) + 'â˜†'.repeat(5 - n);
                }

                function loadTestimonials() {
                    try {
                        const raw = localStorage.getItem(STORAGE_KEY);
                        return raw ? JSON.parse(raw) : [
                            { name: 'Rene Aguinaldo', rating: 5, review: 'Iâ€™ve been ordering from MILLANES Store for months now, and their service is always fast and reliable. The drinks are always fresh and well-stocked. Highly recommended!' },
                            { name: 'Elsi Fuentes', rating: 4, review: 'MILLANES Store never disappoints! Their prices are fair, and they always deliver on time. Perfect for anyone who wants trusted and quality beverages' }
                        ];
                    } catch (e) {
                        return [];
                    }
                }

                function saveTestimonials(data) {
                    try { localStorage.setItem(STORAGE_KEY, JSON.stringify(data)); } catch (e) { }
                }

                function renderTestimonials() {
                    const items = loadTestimonials();
                    list.innerHTML = items.map(it => `
                        <article class="testimonial-card">
                            <div class="testimonial-head">
                                <strong class="t-name">${escapeHtml(it.name)}</strong>
                                <span class="t-stars" aria-hidden="true">${renderStars(it.rating)}</span>
                            </div>
                            <p class="t-text">${escapeHtml(it.review)}</p>
                        </article>
                    `).join('');
                }

                function escapeHtml(s) {
                    return String(s).replace(/[&<>"']/g, (m) => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;' }[m]));
                }

                form.addEventListener('submit', function (e) {
                    e.preventDefault();
                    const name = form['name'].value.trim();
                    const rating = form['rating'].value;
                    const review = form['review'].value.trim();
                    if (!name || !rating || !review) return;
                    const items = loadTestimonials();
                    items.unshift({ name, rating: Number(rating), review, date: new Date().toISOString() });
                    saveTestimonials(items);
                    renderTestimonials();
                    form.reset();
                });

                // initial render
                renderTestimonials();
            })();
        </script>
